  • bind_pure_comp_eq_map : (∀ {α β : Type ?} (f : α → β) (x : m α), x >>= pure ∘ f = f <$> x) . "control_laws_tac"
  • bind_map_eq_seq : (∀ {α β : Type ?} (f : m (α → β)) (x : m α), (f >>= λ (_x : α → β), _x <$> x) = f <*> x) . "control_laws_tac"
